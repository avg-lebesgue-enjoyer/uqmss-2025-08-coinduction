\begin{frame}{Other structures} % .;
  
  \par \strongPink{Every} (non-dependent) inductive \texttt{data} you can make in Haskell/Lean/Agda/etc. has a coinductive version.

  \pause

  \begin{itemize}
    \item[$\bullet$] {
      $\N \rightsquigarrow \coN$
    }\pause
    \item[$\bullet$] {
      $\List \rightsquigarrow \coList$
    }\pause
    \item[$\bullet$] {
      Finite-size binary trees $\rightsquigarrow$ potentially infinite binary trees
    }\pause
    \item[$\bullet$] {
      Etc.
    }
  \end{itemize}

\end{frame}
